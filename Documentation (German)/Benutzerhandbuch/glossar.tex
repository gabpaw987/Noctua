\newglossaryentry{DLL}{
	name=DLL,
	description={Dynamic Link Library (DLL) bezeichnet allgemein eine dynamische Programmbibliothek, meist bezieht sich der Begriff jedoch auf die von Microsoft für die Betriebssysteme Microsoft Windows und OS/2 verwendete Variante.}
}
\newglossaryentry{F-Sharp}{
	name=F\#,
	description={F\# ist eine funktionale Programmiersprache von Microsoft für das .NET-Framework.}
}
\newglossaryentry{BTS}{
	name=Backtesting-Software,
	description={Als BTS wird die im Laufe dieses Projekts erstellte Backtesting-Software bezeichnet, die einen Algorithmus in Form einer DLL-Datei auf seine Performance über historische Aktien-Preisdaten testen kann.}
}
\newglossaryentry{CSV}{
	name=Comma-separated values,
	description={Das Dateiformat CSV steht für englisch Comma-separated values (seltener: Character-separated values) und beschreibt den Aufbau einer Textdatei zur Speicherung oder zum Austausch einfach strukturierter Daten. Die Dateinamenserweiterung lautet .csv.}
}
\newglossaryentry{Bar}{
	name=Bar,
	description={Im Aktienhandel versteht man unter einem Bar die Zusammenfassung aller Preisänderungen eines Instruments über einen festgelegten Zeitraum. Ein Bar wird durch die Werte Open (erster Wert), High (höchster Wert), Low (niedrigster Wert), Close (letzter Wert) innerhalb eines bestimmten Zeitraums definiert.}
}
